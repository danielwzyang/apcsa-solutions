public int sumDigits(String str) {
    // creates an integer to store our sum
    int res = 0;
    
    // loops through every index of the string
    for (int i = 0; i < str.length(); i++) {
        // Character.isDigit() checks to see if a character is a number
        if (Character.isDigit(str.charAt(i))) {
            res += Integer.parseInt(str.substring(i, i + 1));

            // Integer.parseInt() converts a string into a number
            // parseInt only takes strings and charAt returns a character type, so we need to use substring
        }
    }
    return res;
}
